__kernel void simpleMultiply( 
	__global float* a,
	__global float* b,
	__global int* Wa,
	__global int* Wb, 
	__global float* c
	)
{
	int row = get_global_id(0);
	int col = get_global_id(1); 

	float sum = 0.0f; 
	//Calculate result of one element
	for (int i = 0; i < (*Wa); i++)
	{
		sum += a[row * (*Wa) + i] * b[i * (*Wb) + col];
	}

	c[row * (*Wb) + col] = sum;
}
